\subsection{题 1}
证明：以下曲线为弧长参数曲线
$$
\gamma(t)=\left(\frac{(1+t)^{3/2}}{3},\frac{(1-t)^{3/2}}{3},\frac{t}{\sqrt{2}},\right)
$$
其中 $-\frac{1}{2}\le t \le \frac{1}{2}$

证：
$$
\begin{aligned}
\gamma'(t)&=\left(\frac{\sqrt{1+t}}{2},\frac{\sqrt{1-t}}{2},\frac{\sqrt{2}}{2}\right)\\
\|\gamma'(t)\|
&= \sqrt{\frac{1+t}{4} + \frac{1-t}{4} + \frac{1}{2}}\\
&= 1\\
\end{aligned}
$$
故 $\gamma(t)$ 是弧长参数化曲线。

\subsection{题 2}
计算一下两条参数曲线的曲率 $\kappa(t)$,
$$
\begin{aligned}
c_1(t)&=(t,t^2)\\
c_2(t)&=(\cos t, t, \sin t)\\
\end{aligned}
$$

解：
$$
\kappa(t)=\frac{\|c'(t)\times c''(t)\|}{\|c'(t)\|^3}
$$

(1) $c_1(t)=(t,t^2)$
$$
\begin{aligned}
c_1'(t) &= (1, 2t)\\
c_1''(t) &= (0, 2)\\
c_1'(t)\times c_1''(t) &= 2\\
\kappa(t)&=\frac{2}{(4t^2+1)^{\frac{3}{2}}}
\end{aligned}
$$

(2) $c_2(t)=(\cos t, t, \sin t)$
$$
\begin{aligned}
c_2'(t) &= (-\sin t, 1, \cos t)\\
c_2''(t) &= (-\cos t, 0, -\sin t)\\
c_2'(t)\times c_2''(t) &= (-\sin t, -1, \cos t)\\
\kappa(t)&=\frac{1}{2}
\end{aligned}
$$

\subsection{题 3}
证明：以下曲线是平面曲线，
$$
c(t)=(\frac{1+t^2}{t},t+1,\frac{1-t}{t})
$$

证：
$$
\begin{aligned}
\pmb{n} &= (-1,1,1)\\
c(t) - c(1) &= \left(\frac{(t-1)^2}{t}, (t-1), \frac{1-t}{t}\right)\\
\pmb{n} \cdot (c(t)-c(1))
&= -\frac{(t-1)^2}{t} + (t-1) + \frac{1-t}{t}\\
&= \frac{-(t-1)^2+t(t-1)+1-t}{t}\\
&= 0\\
\end{aligned}
$$
即 $\pmb{n} \perp (c(t)-c(1))$，故 $c(t)$ 是平面曲线

\subsection{题 4}
当半径为 $r$ 的“动圆”沿着半径为 $R$ 的“定圆”的外侧无滑动地滚动时，动圆圆周上一定点 $\pmb{p}$ 所描绘的点的轨迹，叫做外摆线。计算外摆线的参数曲线，并画出当 $r=1$，$R=3$ 时的曲线形状

解：
记定圆圆心为 $\pmb{o}_1=(0,0)$，半径为 $R$，动圆圆心为 $\pmb{o}_2(t)$，半径为 $r$。定点 $\pmb{p}(t)$ 与轴线的夹角（逆时针方向）为 $\theta(t)$。初始时 $\pmb{o}_2(0)=(R+r,0),\theta(0)=\theta_0$，滚动速度为 $\nu$，滚动一段时间 $t$ 后，轴线角度为 $\varphi(t)$。如下图所示
\begin{figure}[H]
  \centering
  \includegraphics[width=1.0\textwidth]{p4_0}
  \caption{外摆线示意图}
\end{figure}
$$
\begin{aligned}
\phi(t) &= \frac{\nu t}{R}\\
\theta(t) &= \theta_0 + \frac{\nu t}{r}\\
\pmb{o}_2(t)&=((R+r)\cos \phi(t),(R+r)\sin \phi(t))\\
\pmb{p}(t)
&=\pmb{o}_2(t)+(r\cos (\theta(t)+\phi(t)), r \sin (\theta(t)+\phi(t)))\\
&=\left(
(R+r)\cos \frac{\nu t}{R} + r\cos \left(\theta_0 + \frac{\nu t}{r}+\frac{\nu t}{R}\right),
(R+r)\sin \frac{\nu t}{R} + r\sin \left(\theta_0 + \frac{\nu t}{r}+\frac{\nu t}{R}\right)
\right)
\end{aligned}
$$

\section{编程作业}
\subsection{简述}
渐屈线是曲线上密切圆圆心的轨迹。特别的，Frenet 标架为 $\{e_1(t),e_2(t)\}$ 的平面 Frenet 曲线$c:D\mapsto \mathbb{R}^2$ 可由以下参数曲线 $\eta:D\mapsto \mathbb{R}^2$ 表示，
$$
\eta(t)=c(t)+\frac{1}{\kappa(t)}e_2(t)
$$
编写程序画出椭圆的渐屈线及下图中标记点的密切圆。
\begin{figure}[H]
  \centering
  \includegraphics[width=0.2125\textwidth]{p5_0}
  \caption{渐屈线示意图}
\end{figure}
\subsection{推导}
记长半轴为 $a$，短半轴为 $b$，则
$$
\begin{aligned}
c(t)&=(a\cos t, b\sin t)\\
\dot{c}(t)&=(-a\sin t, b\cos t)\\
\ddot{c}(t)&=(-a\cos t,-b\sin t)\\
e_1(t)&=\frac{\dot{c}(t)}{\|\dot{c}(t)\|}=\frac{(-a\sin t, b\cos t)}{\sqrt{a^2\sin^2 t+b^2 \cos^2 t}}\\
e_1'(t) &= \frac{\ddot{c}(t)}{\|\dot{c}(t)\|^2}=\frac{(-a\cos t,-b\sin t)}{a^2\sin^2 t+b^2 \cos^2 t}\\
e_2(t)&=R^{90^\circ}e_1(t)=\frac{(-b\cos t, -a\sin t)}{\sqrt{a^2\sin^2 t+b^2 \cos^2 t}}\\
\kappa(t)&=e_1'(t)\cdot e_2(t)=\frac{ab}{(a^2\sin^2 t+b^2 \cos^2 t)^{\frac{3}{2}}}\\
\eta(t)&=\left(\left(a-\frac{a^2\sin^2 t+b^2 \cos^2 t}{a}\right)\cos t,\left(b-\frac{a^2\sin^2 t+b^2 \cos^2 t}{b}\right)\sin t\right)\\
\end{aligned}
$$
